- name: uninstall apache from remote
  apt:
    name: apache2
    state: absent
  when: install_apache_flag

- name: check OS value
  debug:
    var: ansible_facts['kernel_version']

  command: cat /home/ubuntu/userdetails.txt
  register: data

- name: print the data if user found
  debug:
    var: data.stdout
  when: data.stdout.find('superuser') != -1
