# CLASS
class Employee():
    uname=''
    email=''
    income=''

    # constructor to initialize the object
    def __init__(self,uname,email,income):
        self.uname=uname
        self.email=email
        self.income=income

    def printDetails(self):
        print(f"{self.uname} , {self.email} , {self.income}")

# Object
Manohar=Employee('Manohar CH','manohar@mail.com','123456')
Manohar.printDetails()
